  9def encode_user_symptoms(user_symptoms, all_symptoms):
 10    """
 11    Converts user symptoms into a binary vector based on all possible symptoms.
 12    Returns the encoded vector and a list of unrecognized symptoms.
 13    """
 14    input_vector = np.zeros(len(all_symptoms))
 15    symptom_to_index = {symptom: idx for idx, symptom in enumerate(all_symptoms)}
 16    unrecognized = []
 17
 18    for symptom in user_symptoms:
 19        symptom = symptom.strip().lower()
 20        if symptom in symptom_to_index:
 21            index = symptom_to_index[symptom]
 22            input_vector[index] = 1
 23        else:
 24            unrecognized.append(symptom)
 25
 26    return input_vector.reshape(1, -1), unrecognized

 28def encode_user_symptoms_fromgpt(user_symptoms, all_symptoms):
 29    """
 30    Encodes the user-extracted symptoms into a binary vector based on the list of all symptoms.
 31    
 32    :param user_symptoms: List of symptoms extracted from user input.
 33    :param all_symptoms: List of all possible symptoms.
 34    :return: Numpy array representing the encoded symptoms.
 35    """
 36    encoded = [1 if symptom.lower() in [s.lower() for s in user_symptoms] else 0 for symptom in all_symptoms]
 37    return np.array([encoded])

295def find_match(input_text, embeddings_model, index, faiss_store, top_k=2):
296    """
297    Finds the closest matches for the given input using the FAISS index.
298    
299    Args:
300        input_text (str): The user input text to match.
301        embeddings_model: The embeddings model to use.
302        index: The FAISS index used for similarity search.
303        faiss_store: The LangChain FAISS store containing the metadata.
304        top_k (int): Number of top matches to retrieve.
305    
306    Returns:
307        str: The combined metadata text from the top matches.
308    """
309    # Encode the input text to create an embedding
310    input_embedding = np.array([embeddings_model.embed_query(input_text)])  # Embed input
311    # Search the FAISS index for the closest matches
312    distances, indices = index.search(input_embedding, top_k)
313    # Retrieve metadata from the FAISS store for the top matches
314    matches = []
315    for i in range(top_k):
316        if indices[0][i] != -1:
317            document = faiss_store.docstore.search(indices[0][i])
318            if document and hasattr(document, 'page_content'):
319                matches.append(document.page_content)
320    # Combine the matches' content
321    result = "\n".join(matches)
322    return result

def find_match(input_text, embeddings_model, index, faiss_store, top_k=2): 
296def find_match(input_text, embeddings_model, index, faiss_store, top_k=2):
297    """
298    Finds the closest matches for the given input using the FAISS index.
299    
300    Args:
301        input_text (str): The user input text to match.
302        embeddings_model: The embeddings model to use.
303        index: The FAISS index used for similarity search.
304        faiss_store: The LangChain FAISS store containing the metadata.
305        top_k (int): Number of top matches to retrieve.
306    
307    Returns:
308        str: The combined metadata text from the top matches.
309    """
310    # Encode the input text to create an embedding
311    input_embedding = np.array([embeddings_model.embed_query(input_text)])  # Embed input
312    # Search the FAISS index for the closest matches
313    distances, indices = index.search(input_embedding, top_k)
314    # Retrieve metadata from the FAISS store for the top matches
315    matches = []
316    for i in range(top_k):
317        if indices[0][i] != -1:
318            document = faiss_store.docstore.search(indices[0][i])
319            if document and hasattr(document, 'page_content'):
320                matches.append(document.page_content)
321    # Combine the matches' content
322    result = "\n".join(matches)
323    return result

Args: input_text (str): The user input text to match. embeddings_model: The embeddings model to use. index: The FAISS index used for similarity search. faiss_store: The LangChain FAISS store containing the metadata. top_k (int): Number of top matches to retrieve.

Returns: str: The combined metadata text from the top matches.

326    """
327    Converts a string representation of a list into an actual list.
328
329    Args:
330        s (str): The string representation of the list.
331
332    Returns:
333        list: The converted list.
334    """
335    # Remove the square brackets if present
336    s = s.strip('[]')
337
338    # Split the string by commas
339    items = s.split(',')
340
341    # Strip whitespace and quotes from each item
342    items = [item.strip(" '\"") for item in items]

344    return items
